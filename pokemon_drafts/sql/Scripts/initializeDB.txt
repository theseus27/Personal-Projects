DROP DATABASE IF EXISTS pokemonDB;
CREATE DATABASE IF NOT EXISTS pokemonDB;
USE pokemonDB;
DROP TABLE IF EXISTS Pokemon;
CREATE TABLE Pokemon(
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(50) NOT NULL,
imgURL VARCHAR(99),
type1 VARCHAR(15) NOT NULL,
type2 VARCHAR(15));
DROP TABLE IF EXISTS BaseStats;
CREATE TABLE BaseStats(
id INT AUTO_INCREMENT PRIMARY KEY,
exp int NOT NULL,
hp int NOT NULL,
att int NOT NULL,
def int NOT NULL,
spAtt int NOT NULL,
spDef int NOT NULL,
speed int NOT NULL,
hpEV int NOT NULL,
attEV int NOT NULL,
defEV int NOT NULL,
spAttEV int NOT NULL,
spDefEV int NOT NULL,
speedEV int NOT NULL);
